#[cfg(test)]
mod tests {
  use super::*;
  #[test]
  fn test_saturation_of_assets() {
    assert_eq!(Asset::Bitcoin.saturating_mul(10), 1_000_000_000);
    assert_eq!(Asset::Tdfy.saturating_mul(912), 912_000_000_000_000);
    assert_eq!(
      Asset::USDCoin.saturating_mul(838_912_012),
      838_912_012_000_000
    );
  }
}
