/// Handles
pub const TC_H_MAJ_MASK: u32 = 0xFFFF0000;
pub const TC_H_MIN_MASK: u32 = 0x0000FFFF;

#[macro_export]
macro_rules! TC_H_MAKE {
    ($maj: expr, $min: expr) => {
        ($maj & TC_H_MAJ_MASK) | ($min & TC_H_MIN_MASK)
    };
}
