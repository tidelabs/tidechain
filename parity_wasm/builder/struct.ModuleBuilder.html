Import entry builder

Examples

let module = module()
   .import()
       .module("env")
       .field("memory")
       .external().memory(256, Some(256))
       .build()
   .build();

assert_eq!(module.import_section().expect("import section to exist").entries().len(), 1);

Examples

   .global()

        .build()
   .export()
       .field("_zero")

       .build()
